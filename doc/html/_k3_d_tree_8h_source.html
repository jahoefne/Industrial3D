47  KDNode *recConstruct(Point3D *begin, Point3D *end, int depth) {
48  unsigned long numPoints = (end - begin);
49  unsigned long medianPosition = numPoints / 2;
50  double medianValue = 0;
51 
52  switch (depth % 3) {
53  case X_AXIS:
54  std::sort(begin, end, [](const Point3D &a, const Point3D &b) -> bool { return a.x < b.x; });
55  medianValue = (begin + medianPosition)->x;
56  break;
57  case Y_AXIS:
58  std::sort(begin, end, [](const Point3D &a, const Point3D &b) -> bool { return a.y < b.y; });
59  medianValue = (begin + medianPosition)->y;
60  break;
61  case Z_AXIS:
62  std::sort(begin, end, [](const Point3D &a, const Point3D &b) -> bool { return a.z < b.z; });
63  medianValue = (begin + medianPosition)->z;
64  break;
65  default:
66  printf("Can not happen!");
67  break;
68  }
69 
70  KDNode *child = new KDNode;
71  child->median = medianValue;
72  child->ptrFirstPoint = begin;
73  child->ptrLastPoint = end;
74 
75  if (numPoints > 1) {
76  child->left = this->recConstruct(begin, begin + medianPosition, depth + 1);
77  child->right = this->recConstruct(begin + medianPosition, end, depth + 1);
78  }
79  return child;
80  }

132  void findRadiusNeighborsRec(KDNode* node, Point3D* queryPoint, double radius, std::vector<Point3D*>* neighbors, unsigned int depth)
133  {
134  if (node->left == node->right) // if this is true the node is a leaf
135  {
136  Point3D* pt = node->ptrFirstPoint;
137 
138  // We don't use the expensive sqrt-function but instead compare the squared values
139  const double sqDistance = pt->sqDistance3d(queryPoint); // squared distance
140  if (sqDistance > (radius*radius)) return; // point outside the sphere
141  neighbors->push_back(node->ptrFirstPoint);
142  }
143  else
144  {
145  const unsigned int DIM = depth % 3;
146  double v_min, v_max;
147  if (DIM == 0){
148  v_min = queryPoint->x - radius;
149  v_max = queryPoint->x + radius;
150  }
151  else if (DIM == 1){
152  v_min = queryPoint->y - radius;
153  v_max = queryPoint->y + radius;
154  }
155  else{
156  v_min = queryPoint->z - radius;
157  v_max = queryPoint->z + radius;
158  }
159 
160  if (v_min <= node->median) findRadiusNeighborsRec(node->left , queryPoint, radius, neighbors, depth + 1);
161  if (v_max > node->median) findRadiusNeighborsRec(node->right, queryPoint, radius, neighbors, depth + 1);
162  }
163  }
